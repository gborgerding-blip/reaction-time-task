function doPost(e) {
  try {
    // Parse incoming JSON
    const data = JSON.parse(e.postData.contents);

    const participant = data.Participant_ID;
    const trial = data.Trial_Number;
    const rt = data.Reaction_Time;

    // Timestamp
    const timestamp = new Date();

    // Append to active sheet
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    sheet.appendRow([timestamp, participant, trial, rt]);

    return ContentService
      .createTextOutput(JSON.stringify({ status: "success" }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: "error", message: err }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}


function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const data = sheet.getDataRange().getValues();

  // Assuming headers: Timestamp | Participant_ID | Trial_Number | Reaction_Time
  const reactionTimes = [];

  for (let i = 1; i < data.length; i++) {
    reactionTimes.push(data[i][3]); // Column D (index 3)
  }

  return ContentService
    .createTextOutput(JSON.stringify(reactionTimes))
    .setMimeType(ContentService.MimeType.JSON);
}
